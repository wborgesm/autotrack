model Peca {
  id          String   @id @default(cuid())
  tenantId    String
  tenant      Tenant   @relation(fields: [tenantId], references: [id], onDelete: Cascade)

  codigo      String
  nome        String
  descricao   String?
  categoria   String?
  unidade     UnidadeMedida @default(UN)
  qtdEstoque  Decimal       @default(0) @db.Decimal(10,2)
  qtdMinima   Decimal       @default(0) @db.Decimal(10,2)
  custoPadrao Decimal?      @db.Decimal(10,2)
  ativo       Boolean       @default(true)

  movimentos  MovimentoEstoque[]
  itensPeca   ItemPecaOrdem[]

  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@unique([tenantId, codigo])
  @@map("pecas")
}
